function canPlace(grid, x, y, w, h){
    if(x + w > 4 || y + h > 5) return false;

    for(let i = 0; i < h; i++){
        for(let j = 0; j < w; j++){
            if(grid[y + i][x + j]) return false;
        }
    }
    return true;
}

function place(grid, item){
    for(let i = 0; i < item.h; i++){
        for(let j = 0; j < item.w; j++){
            grid[item.y + i][item.x + j] = item.id;
        }
    }
}
